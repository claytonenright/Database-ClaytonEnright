--Create Statements--

CREATE TABLE people (
  pid         char(3) not null,
  firstName   text,
  lastName    text,
  age         integer,
 primary key(pid)
);

CREATE TABLE astronauts (
  pid          char(3) not null references people(pid),
  yearsFlying  integer,
 primary key(pid)
);

CREATE TABLE engineers (
  pid         char(3) not null references people(pid),
  highestDeg  text,
 primary key(pid)
);

CREATE TABLE flightcontrol (
  pid        char(3) not null references people(pid),
  prefChair  integer,
 primary key(pid)
);

CREATE TABLE spacecraft (
  sid         char(3) not null,
  name        text,
  tailNum     integer,
  weightTons  integer,
  fuelType    text,
  crewCap     integer,
 primary key(sid)
);

CREATE TABLE crew (
  pid    char(3) not null references people(pid),
  sid    char(3) not null references spacecraft(sid),
 primary key(pid, sid)
);

CREATE TABLE systems (
  sysid        char(5) not null,
  name         text,
  descrip      text,
 primary key(sysid)
);

CREATE TABLE syssetup (
  sid    char(3) not null references spacecraft(sid),
  sysid  char(5) not null references systems(sysid),
 primary key(sid, sysid)
);

CREATE TABLE parts (
  partid      char(7) not null,
  name        text,
  descrip     text,
 primary key(partid)
);

CREATE TABLE partsetup (
  sysid    char(5) not null references systems(sysid),
  partid   char(7) not null references parts(partid),
 primary key(sysid, partid)
);

CREATE TABLE zipcode (
  zip     char(5) not null,
  state   text,
  city    text,
 primary key(zip)
);

CREATE TABLE suppliers (
  supid       char(5) not null,
  name        text,
  streetNum   integer,
  streetName  text,
  zip         char(5) not null references zipcode(zip),
  payTerms    char(5),
 primary key(supid)
);

CREATE TABLE catalog (
  partid   char(7) not null references parts(partid),
  supid    char(5) not null references suppliers(supid),
 primary key(partid, supid)
);
